macro: "VALUE_COMPONENT\((?<name>\w+),(?<type>(\w|\.)+)\)" 
    => "public class ${name}Component : NaiveEntity.GamePrototype.EntProto.ValueComponent<${type}> { }".
macro: "GET_VALUED_PROPERTY\((?<type>(\w|\.)+),(?<name>\w+)\)" 
    => "public ${type} ${name} { get { return Entity.GetComponent<${name}Component>().Value; } set { Entity.AddComponent(new ${name}Component() { Value = value }); } }".

<entity_define>:
    => $entityType
    => log "Entity " + $entityType,
     create class $entityType,
     add base "NaiveEntity.GamePrototype.EntProto.EntityWrapper" to $entityType,
     set global $entityType + "isEntity" = $entityType.

<typed_component_creation>:
    => $componentName "is" $valueType "property"
    => log "Component " + $componentName + " of " + $valueType,
        add code "VALUE_COMPONENT(" + $componentName + "," + $valueType + ")",
        set global $componentName + "Component" = $componentName,
        set global $componentName + "Value" = $valueType.
